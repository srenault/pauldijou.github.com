RichFaces CDK

A partir de :

  • L'API du composant : Abstract[Nom]
  • Son renderer abstrait : [Nom]RendererBase
  • Son template : [nom].template.xml

Va générer :

  • Le composant : UI[Nom]
  • Son renderer : [Nom]Renderer
  • La taglib des composants : [nom].taglib.xml
  • Toutes les ressources correctement packagées
  • Les fragments nécessaires dans le fichier faces-config.xml

L'API du composant

@JsfComponent(
    type = AbstractLabel.COMPONENT_TYPE,
    family = AbstractLabel.COMPONENT_FAMILY,
    renderer = @JsfRenderer(type = LabelRendererBase.RENDERER_TYPE),
    tag = @Tag(name="label"))
public abstract class AbstractLabel extends UIOutput implements CoreProps {
    public static final String COMPONENT_FAMILY = "org.richfaces.bootstrap.Label";
    public static final String COMPONENT_TYPE = "org.richfaces.bootstrap.Label";
    
    @Attribute
    public abstract String getSeverity(); 
}

Son renderer abstrait

@ResourceDependencies({
  @ResourceDependency(library = "org.richfaces", name = "ajax.reslib"),
  @ResourceDependency(library = "org.richfaces", name = "base-component.reslib"),
  @ResourceDependency(library = "org.richfaces", name = "bootstrap-css.reslib")})
public abstract class LabelRendererBase extends RendererBase {
    public static final String RENDERER_TYPE = "org.richfaces.bootstrap.LabelRenderer";
}

Son template

<cc:implementation>
    <span id="#{clientId}"
        class="label #{component.severity != null ? 'label-'.concat(component.severity) : ''} #{component.attributes['styleClass']}"
        cdk:passThrough="style">
        <c:if test="#{component.value != null and !component.value.equals('')}">#{component.value}</c:if>
        <cdk:body />
    </span>
</cc:implementation>
